 19class MetaConvention(BaseModel):
 20    """
 21    Filename convention for metadata files that are associated with other entities.
 22
 23    It defines where to look for metadata for files that are not themselves known as json,
 24    or metadata concerning directories.
 25
 26    At the same time, these files are ignored by themselves and act as "sidecar" files.
 27    """
 28
 29    pathPrefix: str = ""
 30    pathSuffix: str = ""
 31    filePrefix: str = ""
 32    fileSuffix: str = "_meta.json"
 33
 34    @root_validator
 35    def check_valid(cls, values):
 36        """Check that at least one filename extension is non-empty."""
 37        file_pref_or_suf = values.get("filePrefix", "") or values.get("fileSuffix", "")
 38        assert file_pref_or_suf, "At least one of filePrefix or fileSuffix must be set!"
 39        return values
 40
 41    def to_tuple(self) -> Tuple[str, str, str, str]:
 42        """Convert convention instance to tuple (e.g. used within CLI)."""
 43        return (self.pathPrefix, self.pathSuffix, self.filePrefix, self.fileSuffix)
 44
 45    @classmethod
 46    def from_tuple(cls, pp: str, ps: str, fp: str, fs: str):
 47        """Return new metadata file convention."""
 48        return MetaConvention(
 49            pathPrefix=pp, pathSuffix=ps, filePrefix=fp, fileSuffix=fs
 50        )
 51
 52    def is_meta(self, path: str) -> bool:
 53        """Check whether given path is a metadata file according to the convention."""
 54        prts = Path(path).parts
 55        if len(prts) == 0:  # root dir
 56            return False
 57        if self.filePrefix != "" and not prts[-1].startswith(self.filePrefix):
 58            return False
 59        if self.fileSuffix != "" and not prts[-1].endswith(self.fileSuffix):
 60            return False
 61        pieces = int(self.pathPrefix != "") + int(self.pathSuffix != "")
 62        if len(prts) < 1 + pieces:
 63            return False
 64        pp = self.pathPrefix == "" or prts[0] == self.pathPrefix
 65        ps = self.pathSuffix == "" or prts[-2] == self.pathSuffix
 66        return pp and ps
 67
 68    def meta_for(self, path: str, is_dir: bool = False) -> str:
 69        """Return metadata filename for provided path, based on this convention."""
 70        ps = list(Path(path).parts)
 71        newp = []
 72
 73        if self.pathPrefix != "":
 74            newp.append(self.pathPrefix)
 75        newp += ps[:-1]
 76        if not is_dir and self.pathSuffix != "":
 77            newp.append(self.pathSuffix)
 78        name = ps[-1] if len(ps) > 0 else ""
 79
 80        if is_dir:
 81            newp.append(name)
 82            if self.pathSuffix != "":
 83                newp.append(self.pathSuffix)
 84            metaname = self.filePrefix + self.fileSuffix
 85            newp.append(metaname)
 86        else:
 87            metaname = self.filePrefix + name + self.fileSuffix
 88            newp.append(metaname)
 89        return str(Path().joinpath(*newp))

 92class PathSlice(BaseModel):
 93    """
 94    Helper class to slice into segments of a path, match a regex and perform substitutions.
 95
 96    Invariant: into(path, sl).unslice() == path for all sl and path.
 97    """
 98
 99    slicePre: Optional[str]
100    sliceStr: str
101    sliceSuf: Optional[str]
102
103    @classmethod
104    def into(
105        cls, path: str, start: Optional[int] = None, stop: Optional[int] = None
106    ) -> PathSlice:
107        """
108        Slice into a path, splitting on the slashes.
109
110        Slice semantics is mostly like Python, except that stop=0 means
111        "until the end", so that [0:0] means the full path.
112        """
113        segs = path.split("/")
114        pref = "/".join(segs[: start if start else 0])
115        inner = "/".join(segs[start : stop if stop != 0 else None])  # noqa: E203
116        suf = "/".join(segs[stop:] if stop else [])
117        return PathSlice(
118            slicePre=pref if pref else None,
119            sliceStr=inner,
120            sliceSuf=suf if suf else None,
121        )
122
123    def unslice(self) -> str:
124        """Inverse of slice operation (recovers complete path string)."""
125        return "/".join([x for x in [self.slicePre, self.sliceStr, self.sliceSuf] if x])
126
127    _def_pat = re.compile("(.*)")
128    """Default pattern (match anything, put into capture group)."""
129
130    def match(self, pat: Optional[Union[re.Pattern, str]] = None):
131        pat = pat or self._def_pat
132        if isinstance(pat, str):
133            pat = re.compile(pat)
134        return pat.fullmatch(self.sliceStr)
135
136    def rewrite(
137        self, pat: Optional[Union[re.Pattern, str]] = None, sub: Optional[str] = None
138    ) -> Optional[PathSlice]:
139        """
140        Match and rewrite in the slice string and return a new PathSlice.
141
142        If no pattern given, default pattern is used.
143        If no substitution is given, just match on pattern is performed.
144        Returns new PathSlice with possibly rewritten slice.
145        Returns None if match fails.
146        Raises exception of rewriting fails due to e.g. invalid capture groups.
147        """
148        if m := self.match(pat):
149            ret = self.copy()
150            if sub is not None:
151                ret.sliceStr = m.expand(sub)
152            return ret
153        return None

169class TypeEnum(Enum):
170    """
171    Possible values for a path type inside a dirschema rule.
172
173    MISSING means that the path may not exist (i.e. neither file or directory),
174    whereas ANY means that any of these options is fine, as long as the path exists.
175    """
176
177    MISSING = False
178    FILE = "file"
179    DIR = "dir"
180    ANY = True
181
182    def is_satisfied(self, is_file: bool, is_dir: bool) -> bool:
183        if self == TypeEnum.MISSING and (is_file or is_dir):
184            return False
185        if self == TypeEnum.ANY and not (is_file or is_dir):
186            return False
187        if self == TypeEnum.DIR and not is_dir:
188            return False
189        if self == TypeEnum.FILE and not is_file:
190            return False
191        return True
192
193
194DEF_MATCH: Final[str] = "(.*)"
195"""Default match regex to assume when none is set, but required by semantics."""
196
197DEF_REWRITE: Final[str] = "\\1"
198"""Default rewrite rule to assume when none is set, but required by semantics."""

201class DSRule(BaseModel):
202    """
203    A DirSchema rule is either a trivial (boolean) rule, or a complex object.
204
205    Use this class for parsing, if it is not known which of these it is.
206    """
207
208    __root__: Union[bool, Rule]
209
210    def __init__(self, b: Optional[bool] = None, **kwargs):
211        """Construct wrapped boolean or object, depending on arguments."""
212        if b is not None:
213            return super().__init__(__root__=b)
214        elif "__root__" in kwargs:
215            assert len(kwargs) == 1, "No extra kwargs may be passed with __root__!"
216            return super().__init__(**kwargs)
217        else:
218            return super().__init__(__root__=Rule(**kwargs))
219
220    def __repr__(self) -> str:
221        """Make wrapper transparent and just return repr of wrapped object."""
222        if isinstance(self.__root__, bool):
223            return "true" if self.__root__ else "false"
224        else:
225            return repr(self.__root__)
226
227    def __bool__(self):
228        """Just return value for wrapped object."""
229        return bool(self.__root__)

232class Rule(BaseModel):
233    """A DirSchema is a conjunction of at most one of each possible constraint/keyword."""
234
235    # primitive:
236    type: Optional[TypeEnum] = Field(
237        description="Check that path is a file / is a dir."
238    )
239
240    valid: Optional[Union[JSONSchema, str]] = Field(
241        description="Validate file against provided schema or validator."
242    )
243
244    # this will use the provided metadataConvention for rewriting to the right path
245    validMeta: Optional[Union[JSONSchema, str]] = Field(
246        description="Validate external metadata against provided schema or validator."
247    )
248
249    # these are JSON-Schema-like logical operators:
250
251    allOf: List[DSRule] = Field([], description="Conjunction (evaluated in order).")
252
253    anyOf: List[DSRule] = Field([], description="Disjunction (evaluated in order).")
254
255    oneOf: List[DSRule] = Field([], description="Exact-1-of-N (evaluated in order).")
256
257    not_: Optional[DSRule] = Field(description="Negation of a rule.", alias="not")
258
259    # introduced for better error reporting (will yield no error message on failure)
260    # So this is more for dirschema "control-flow"
261    if_: Optional[DSRule] = Field(
262        description="Depending on result of rule, will proceed with 'then' or 'else'.",
263        alias="if",
264    )
265
266    then: Optional[DSRule] = Field(
267        description="Evaluated if 'if' rule exists and satisfied.",
268    )
269
270    else_: Optional[DSRule] = Field(
271        description="Evaluated if 'if' rule exists and not satisfied.",
272        alias="else",
273    )
274
275    # match and rewrite (path inspection and manipulation):
276
277    # we keep the total match and the capture groups for possible rewriting
278    # the match data + start/end is also inherited down into children
279    match: Optional[Pattern] = Field(
280        description="Path must match. Sets capture groups."
281    )
282
283    # indices of path segments (i.e. array parts after splitting on /)
284    # matchStart < matchEnd (unless start pos. and end neg.)
285    # matchStart = -1 = match only in final segment
286    # it's python slice without 'step' option
287    # this means, missing segments are ignored
288    # to have "exact" number of segments, match on a pattern with required # of / first
289    matchStart: Optional[int]
290    matchStop: Optional[int]
291
292    # only do rewrite if match was successful
293    rewrite: Optional[str]
294
295    # if rewrite is set, apply 'next' to rewritten path instead of original
296    # missing rewrite is like rewrite \1, missing match is like ".*"
297    next: Optional[DSRule] = Field(
298        description="If current rule is satisfied, evaluate the 'next' rule."
299    )
300
301    # improve error reporting by making it customizable
302    # overrides all other errors on the level of this rule (but keeps subrule errors)
303    # if set to "", will report no error message for this rule
304    description: Optional[str] = Field(
305        None,
306        description="Custom error message to be displayed to the user if this rule fails.",
307    )
308    # used to prune noisy error message accumulation to some high level summary
309    details: bool = Field(
310        True,
311        description="If set, keep errors from sub-rules, otherwise ignore them.",
312    )
313
314    # ----
315
316    def __repr__(self, stream=None) -> str:
317        """Print out the rule as YAML (only the non-default values)."""
318        res = json.loads(self.json(exclude_defaults=True))
319
320        if not stream:
321            stream = io.StringIO()
322            yaml.dump(res, stream)
323            return stream.getvalue().strip()
324
325        yaml.dump(res, stream)
326        return ""

170class TypeEnum(Enum):
171    """
172    Possible values for a path type inside a dirschema rule.
173
174    MISSING means that the path may not exist (i.e. neither file or directory),
175    whereas ANY means that any of these options is fine, as long as the path exists.
176    """
177
178    MISSING = False
179    FILE = "file"
180    DIR = "dir"
181    ANY = True
182
183    def is_satisfied(self, is_file: bool, is_dir: bool) -> bool:
184        if self == TypeEnum.MISSING and (is_file or is_dir):
185            return False
186        if self == TypeEnum.ANY and not (is_file or is_dir):
187            return False
188        if self == TypeEnum.DIR and not is_dir:
189            return False
190        if self == TypeEnum.FILE and not is_file:
191            return False
192        return True

202class DSRule(BaseModel):
203    """
204    A DirSchema rule is either a trivial (boolean) rule, or a complex object.
205
206    Use this class for parsing, if it is not known which of these it is.
207    """
208
209    __root__: Union[bool, Rule]
210
211    def __init__(self, b: Optional[bool] = None, **kwargs):
212        """Construct wrapped boolean or object, depending on arguments."""
213        if b is not None:
214            return super().__init__(__root__=b)
215        elif "__root__" in kwargs:
216            assert len(kwargs) == 1, "No extra kwargs may be passed with __root__!"
217            return super().__init__(**kwargs)
218        else:
219            return super().__init__(__root__=Rule(**kwargs))
220
221    def __repr__(self) -> str:
222        """Make wrapper transparent and just return repr of wrapped object."""
223        if isinstance(self.__root__, bool):
224            return "true" if self.__root__ else "false"
225        else:
226            return repr(self.__root__)
227
228    def __bool__(self):
229        """Just return value for wrapped object."""
230        return bool(self.__root__)
